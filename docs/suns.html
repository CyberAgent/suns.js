Example
var FooClass = function FooClass() {
};
FooClass.prototype = {
    func1: function func1() {}
};
FooClass.extend = suns.extendThis;

// OK
var Child = FooClass.extend({
    childFunc: function childFunc() {}
});
// same as
Child = suns.extend(FooClass, {
    childFunc: function childFunc() {}
});
